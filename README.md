    <script>
        const answersKey = { q1: "B", q2: "B", q3: "A", q4: "B" };

        function evaluateExam() {
            let marks = 0;
            const total = Object.keys(answersKey).length;
            const examData = new FormData(document.getElementById('gbExamPortal'));

            for(let key in answersKey) {
                if(examData.get(key) === answersKey[key]) marks++;
            }

            const finalPercent = Math.round((marks / total) * 100);
            document.getElementById('scoreLabel').innerText = finalPercent + "%";
            document.getElementById('msg').innerText = finalPercent >= 50 ? "Excellent Performance!" : "Needs Improvement.";
            document.getElementById('resultOverlay').classList.remove('hidden');
        }

        let totalTime = 5400;
        setInterval(() => {
            if(totalTime <= 0) return;
            totalTime--;
            let mins = Math.floor(totalTime / 60);
            let secs = totalTime % 60;
            document.getElementById('timer').innerText = `${mins}:${secs < 10 ? '0' : ''}${secs}`;
        }, 1000);
    </script>
